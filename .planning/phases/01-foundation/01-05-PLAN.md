**src/app/(dashboard)/organizations/actions.ts** — Server Actions:

Create with 'use server' at top. Each action must:
1. Call `const supabase = await createClient()` (RLS-scoped user client)
2. Perform the database operation
3. Call `revalidatePath('/dashboard/organizations')` on success
4. Return `{ error: string }` on failure, nothing on success

Slug generation for createOrganization: derive slug from name automatically if not provided:
```typescript
function generateSlug(name: string): string {
  return name.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/(^-|-$)/g, '')
}
```

Error handling: catch Supabase errors and map them:
- PostgreSQL error code '23505' (unique violation) → "An organization with this name already exists."
- Generic fallback: use error.message

```typescript
'use server'
import { createClient } from '@/lib/supabase/server'
import { revalidatePath } from 'next/cache'

function generateSlug(name: string): string {
  return name.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/(^-|-$)/g, '')
}

export async function createOrganization(data: { name: string }) {
  const supabase = await createClient()
  const slug = generateSlug(data.name)
  const { error } = await supabase
    .from('organizations')
    .insert({ name: data.name, slug })
  if (error) {
    if (error.code === '23505') return { error: 'An organization with this name already exists.' }
    return { error: error.message }
  }
  revalidatePath('/dashboard/organizations')
}

export async function updateOrganization(id: string, data: { name: string; is_active: boolean }) {
  const supabase = await createClient()
  const slug = generateSlug(data.name)
  const { error } = await supabase
    .from('organizations')
    .update({ name: data.name, slug, is_active: data.is_active })
    .eq('id', id)
  if (error) return { error: error.message }
  revalidatePath('/dashboard/organizations')
}

export async function toggleOrganizationStatus(id: string, is_active: boolean) {
  const supabase = await createClient()
  const { error } = await supabase
    .from('organizations')
    .update({ is_active })
    .eq('id', id)
  if (error) return { error: error.message }
  revalidatePath('/dashboard/organizations')
}
```

import { createClient } from '@/lib/supabase/server'
import { redirect } from 'next/navigation'
import { OrganizationsTable } from '@/components/organizations/organizations-table'
import { Button } from '@/components/ui/button'

export default async function OrganizationsPage() {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) redirect('/login')

  const { data: organizations, error } = await supabase
    .from('organizations')
    .select('*')
    .order('created_at', { ascending: false })

  if (error) throw new Error(error.message) // Caught by error boundary

  return (
    <div className="p-6">
      <div className="flex items-center justify-between mb-6">
        <div>
          <h1 className="text-xl font-semibold">Organizations</h1>
          <p className="text-sm text-muted-foreground">Manage tenants and their Vapi assistant mappings.</p>
        </div>
        {/* Create button is inside OrganizationsTable to access Sheet state */}
      </div>
      <OrganizationsTable organizations={organizations ?? []} />
    </div>
  )
}
